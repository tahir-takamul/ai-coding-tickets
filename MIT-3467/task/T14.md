### Step 4 — Reusable Jenkins module `[SIBLING] cicd/pipelines/modules/sonar-scan.Jenkinsfile`

```groovy
/**
 * Module: Sonar Scan
 * Runs sonar-scanner / ./gradlew sonar / npm scan for one of backend / portal / wallet
 * against the in-cluster SonarQube at http://sonarqube.cicd.svc:9000 (assuming
 * fullnameOverride: sonarqube — adjust to sonarqube-sonarqube otherwise).
 *
 * Parameters:
 *   - COMPONENT: backend | portal | wallet
 *   - COMMIT_SHA: commit to check out (informational; pipeline runs on whatever was checked out)
 *   - BRANCH: branch name (informational on Community Build — F16)
 *
 * Credential expected: 'sonar-token' (from KV cicd-sonarqube-token, T10).
 */
pipeline {
    agent {
        kubernetes {
            yaml '''
apiVersion: v1
kind: Pod
metadata:
  labels:
    azure.workload.identity/use: "true"
spec:
  serviceAccountName: jenkins
  containers:
  - name: agent
    image: mithrilacr.azurecr.io/cicd-agent:<TAG_FROM_T07>
    command: [cat]
    tty: true
    resources:
      requests: { cpu: 500m, memory: 3Gi }
      limits:   { cpu: 2000m, memory: 6Gi }
'''
        }
    }

    options {
        timestamps()
        timeout(time: 25, unit: 'MINUTES')
    }

    parameters {
        choice(name: 'COMPONENT',  choices: ['backend', 'portal', 'wallet'], description: 'Module to scan')
        string(name: 'COMMIT_SHA', defaultValue: '',     description: 'Commit to check out')
        string(name: 'BRANCH',     defaultValue: 'main', description: 'Branch name (informational; Community Build stores main only)')
    }

    environment {
        SONAR_HOST_URL = 'http://sonarqube.cicd.svc:9000'   // in-cluster Service DNS — F06
    }

    stages {
        stage('Checkout') {
            steps {
                script {
                    if (params.COMMIT_SHA?.trim()) {
                        checkout([$class: 'GitSCM',
                                  branches: [[name: params.COMMIT_SHA]],
                                  userRemoteConfigs: [[url: scm.userRemoteConfigs[0].url]]])
                    } else {
                        checkout scm
                    }
                }
            }
        }

        stage('Scan') {
            steps {
                withCredentials([string(credentialsId: 'sonar-token', variable: 'SONAR_TOKEN')]) {
                    container('agent') {
                        script {
                            // Use sh """ ... \$SONAR_TOKEN """ — preserves Jenkins credential masking.
                            // Never inline ${env.SONAR_TOKEN} into the script string — that prints it.
                            switch (params.COMPONENT) {
                                case 'backend':
                                    dir('backend') {
                                        sh """
                                            ./gradlew clean test jacocoTestReport
                                            ./gradlew sonar \\
                                              -Dsonar.host.url=${env.SONAR_HOST_URL} \\
                                              -Dsonar.token=\$SONAR_TOKEN \\
                                              -Dsonar.qualitygate.wait=true
                                        """
                                    }
                                    break
                                case 'portal':
                                    dir('portal') {
                                        sh """
                                            sonar-scanner \\
                                              -Dsonar.host.url=${env.SONAR_HOST_URL} \\
                                              -Dsonar.token=\$SONAR_TOKEN \\
                                              -Dsonar.qualitygate.wait=true
                                        """
                                    }
                                    break
                                case 'wallet':
                                    dir('wallet') {
                                        sh """
                                            npm ci
                                            npm run test:coverage
                                            sonar-scanner \\
                                              -Dsonar.host.url=${env.SONAR_HOST_URL} \\
                                              -Dsonar.token=\$SONAR_TOKEN \\
                                              -Dsonar.qualitygate.wait=true
                                        """
                                    }
                                    break
                            }
                        }
                    }
                }
            }
        }
    }
}
```

The `cicd-agent` image already carries `sonar-scanner` (T07). Backend's `./gradlew sonar` requires the plugin (Step 3 of this task).
