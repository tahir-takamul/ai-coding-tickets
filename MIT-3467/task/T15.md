## Output

### `[SIBLING] cicd/pipelines/sonar-pr.Jenkinsfile`

```groovy
/**
 * Sonar PR Scan — webhook-triggered from GitHub Actions.
 * GHA POSTs to /generic-webhook-trigger/invoke?token=mithril-sonar-pr-token on PR open/sync.
 * This job validates inputs and dispatches to the sonar-scan module from T14.
 * Build status (SUCCESS/FAILURE) is the quality-gate signal for the PR check.
 */
pipeline {
    agent none

    parameters {
        string(name: 'COMPONENT',  defaultValue: 'backend', description: 'backend | portal | wallet')
        string(name: 'COMMIT_SHA', defaultValue: '',        description: 'PR head commit SHA')
        string(name: 'PR_NUMBER',  defaultValue: '',        description: 'GitHub PR number')
        string(name: 'PR_BRANCH',  defaultValue: 'main',    description: 'PR source branch')
    }

    options {
        timestamps()
        timeout(time: 30, unit: 'MINUTES')
    }

    stages {
        stage('Validate') {
            steps {
                script {
                    if (!(params.COMPONENT in ['backend', 'portal', 'wallet'])) {
                        error "Unsupported COMPONENT: ${params.COMPONENT}"
                    }
                    if (!params.COMMIT_SHA?.trim()) {
                        error "COMMIT_SHA is required"
                    }
                }
            }
        }

        stage('Dispatch') {
            steps {
                // Job path follows the convention used by other modular jobs.
                // Confirm with Jenkins owner when registering this job in JCasC.
                build job: '/mithril/modular-jobs/sonar-scan',
                    parameters: [
                        string(name: 'COMPONENT',  value: params.COMPONENT),
                        string(name: 'COMMIT_SHA', value: params.COMMIT_SHA),
                        string(name: 'BRANCH',     value: params.PR_BRANCH)
                    ],
                    wait: true,
                    propagate: true
            }
        }
    }

    post {
        always {
            echo "Sonar PR scan: COMPONENT=${params.COMPONENT} PR=#${params.PR_NUMBER} SHA=${params.COMMIT_SHA} RESULT=${currentBuild.currentResult}"
        }
    }
}
```

### JCasC update (Azure Jenkins config repo)

Register a new pipeline job `mithril/modular-jobs/sonar-pr` pointing at this Jenkinsfile, with a `generic-webhook-trigger`:

```yaml
- name: 'mithril-sonar-pr-token'
  token: 'mithril-sonar-pr-token'
  genericVariables:
    - { key: COMPONENT,  expressionType: JSONPath, value: $.component }
    - { key: COMMIT_SHA, expressionType: JSONPath, value: $.commit_sha }
    - { key: PR_NUMBER,  expressionType: JSONPath, value: $.pr_number }
    - { key: PR_BRANCH,  expressionType: JSONPath, value: $.pr_branch }
  causeString: 'Triggered by GHA for PR#$PR_NUMBER component=$COMPONENT'
  printContributedVariables: false
  printPostContent: false
```
